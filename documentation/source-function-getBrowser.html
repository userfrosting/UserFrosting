 96:     foreach ($lines as $line){

115: // Useful for testing output of API functions
116: function prettyPrint( $json )
117: {
118:     $result = '';
119:     $level = 0;
120:     $in_quotes = false;
121:     $in_escape = false;
122:     $ends_line_level = NULL;
123:     $json_length = strlen( $json );
124: 
125:     for( $i = 0; $i < $json_length; $i++ ) {
126:         $char = $json[$i];
127:         $new_line_level = NULL;
128:         $post = "";
129:         if( $ends_line_level !== NULL ) {
130:             $new_line_level = $ends_line_level;
131:             $ends_line_level = NULL;
132:         }
133:         if ( $in_escape ) {
134:             $in_escape = false;
135:         } else if( $char === '"' ) {
136:             $in_quotes = !$in_quotes;
137:         } else if( ! $in_quotes ) {
138:             switch( $char ) {
139:                 case '}': case ']':
140:                     $level--;
141:                     $ends_line_level = NULL;
142:                     $new_line_level = $level;
143:                     break;
144: 
145:                 case '{': case '[':
146:                     $level++;
147:                 case ',':
148:                     $ends_line_level = $level;
149:                     break;
150: 
151:                 case ':':
152:                     $post = " ";
153:                     break;
154: 
155:                 case " ": case "\t": case "\n": case "\r":
156:                     $char = "";
157:                     $ends_line_level = $new_line_level;
158:                     $new_line_level = NULL;
159:                     break;
160:             }
161:         } else if ( $char === '\\' ) {
162:             $in_escape = true;
163:         }
164:         if( $new_line_level !== NULL ) {
165:             $result .= "\n".str_repeat( "\t", $new_line_level );
166:         }
167:         $result .= $char.$post;
168:     }

269: // Master function for validating passwords.  Ensures backwards compatibility with sha1 (usercake) and the old homegrown implementation of crypt
270: function passwordVerifyUF($password, $hash){
271:     if (getPasswordHashTypeUF($hash) == "sha1"){
272:         $salt = substr($hash, 0, 25);       // Extract the salt from the hash
273:         $hash_input = $salt . sha1($salt . $password);
274:         if ($hash_input == $hash){
275:             return true;
276:         } else {
277:             return false;
278:         }
279:     }   
280:     // Homegrown implementation (assuming that current install has been using a cost parameter of 12)
281:     else if (getPasswordHashTypeUF($hash) == "homegrown"){
282:         /*used for manual implementation of bcrypt*/
283:         $cost = '12'; 
284:         if (substr($hash, 0, 60) == crypt($password, "$2y$".$cost."$".substr($hash, 60))){
285:             return true;
286:         } else {
287:             return false;
288:         }
289:     // Modern implementation
290:     } else {
291:         return password_verify($password, $hash);
292:     }
293: }

494: /**
495: * array_merge_recursive does indeed merge arrays, but it converts values with duplicate
496: * keys to arrays rather than overwriting the value in the first array with the duplicate
497: * value in the second array, as array_merge does. I.e., with array_merge_recursive,
498: * this happens (documented behavior):
499: *
500: * array_merge_recursive(array('key' => 'org value'), array('key' => 'new value'));
501: *     => array('key' => array('org value', 'new value'));
502: *
503: * array_merge_recursive_distinct does not change the datatypes of the values in the arrays.
504: * Matching keys' values in the second array overwrite those in the first array, as is the
505: * case with array_merge, i.e.:
506: *
507: * array_merge_recursive_distinct(array('key' => 'org value'), array('key' => 'new value'));
508: *     => array('key' => array('new value'));
509: *
510: * Parameters are passed by reference, though only for performance reasons. They're not
511: * altered by this function.
512: *
513: * @param array $array1
514: * @param array $array2
515: * @return array
516: * @author Daniel <daniel (at) danielsmedegaardbuus (dot) dk>
517: * @author Gabriel Sobrinho <gabriel (dot) sobrinho (at) gmail (dot) com>
518: */
519: function array_merge_recursive_distinct ( array &$array1, array &$array2 )
520: {
521:   $merged = $array1;
522: 
523:   foreach ( $array2 as $key => &$value )
524:   {
525:     if ( is_array ( $value ) && isset ( $merged [$key] ) && is_array ( $merged [$key] ) )
526:     {
527:       $merged [$key] = array_merge_recursive_distinct ( $merged [$key], $value );
528:     }
529:     else
530:     {
531:       $merged [$key] = $value;
532:     }
533:   }
534: 
535:   return $merged;
536: }
