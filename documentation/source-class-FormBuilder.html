304:     private function generateFieldData($field_name){
305:         $field = $this->_fields[$field_name];
306:         
307:         $field_data = array();
308:         
309:         $field_data['name'] = $field_name;
310:         $field_data['label'] = isset($field['label']) ? $field['label'] : $field_name;
311:         $field_data['placeholder'] = isset($field['placeholder']) ? $field['placeholder'] : "";
312:         
313:         $icon = isset($field['icon']) ? $field['icon'] : "fa fa-edit";
314:         $icon_link = isset($field['icon_link']) ? $field['icon_link'] : null;
315:         if ($icon_link)
316:             $field_data['addon'] = "<a href='$icon_link'><i class='$icon'></i></a>";
317:         else
318:             $field_data['addon'] = "<i class='$icon'></i>";
319:         
320:         $display = isset($field['display']) ? $field['display'] : "show";
321:         if ($display == "hidden"){
322:             $field_data['hidden'] = "hidden";
323:             $field_data['disabled'] = "disabled";
324:         } else if ($display == "disabled"){
325:             $field_data['hidden'] = "";
326:             $field_data['disabled'] = "disabled";
327:         } else {
328:             $field_data['hidden'] = "";
329:             $field_data['disabled'] = "";            
330:         }
331: 
332:         $validator = isset($field['validator']) ? $field['validator'] : array();
333:         $field_data['validator_str'] = json_encode($validator, JSON_FORCE_OBJECT);
334:         
335:         if (isset($this->_data[$field_name]))
336:             $field_data['value'] = $this->_data[$field_name];
337:         else {
338:             // Set default value
339:             if (isset($field['default'])){
340:                 $field_data['value'] = $field['default'];
341:             } else {
342:                 $field_data['value'] = "";
343:             }
344:         }
345:         
346:         // Preprocess value
347:         if (isset($field['preprocess'])){
348:             $method = new ReflectionFunction($field['preprocess']);
349:             if ($method){
350:                 $field_data['value'] = $method->invokeArgs(array($field_data['value']));
351:             }
352:         }
353:         
354:         return $field_data;
355:     }
