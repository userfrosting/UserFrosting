 102: function createUser($user_name, $display_name, $email, $title, $password, $passwordc, $require_activation, $admin) {
 103:     // if we're in admin mode, then the user must be logged in and have appropriate permissions
 104:     if ($admin == "true"){
 105:         // This block automatically checks this action against the permissions database before running.
 106:         if (!checkActionPermissionSelf(__FUNCTION__, func_get_args())) {
 107:             addAlert("danger", "Sorry, you do not have permission to access this resource.");
 108:             return false;
 109:         }
 110:     }
 111: 
 112:     $error_count = 0;
 113: 
 114:     // Check values
 115:     if(minMaxRange(1,25,$user_name))
 116:     {
 117:         addAlert("danger", lang("ACCOUNT_USER_CHAR_LIMIT",array(1,25)));
 118:         $error_count++;
 119:     }
 120:     if(!ctype_alnum($user_name)){
 121:         addAlert("danger", lang("ACCOUNT_USER_INVALID_CHARACTERS"));
 122:         $error_count++;
 123:     }
 124:     if(minMaxRange(1,50,$display_name))
 125:     {
 126:         addAlert("danger", lang("ACCOUNT_DISPLAY_CHAR_LIMIT",array(1,50)));
 127:         $error_count++;
 128:     }
 129:     if(!isValidName($display_name)){
 130:         addAlert("danger", lang("ACCOUNT_DISPLAY_INVALID_CHARACTERS"));
 131:         $error_count++;
 132:     }
 133:     if(!isValidEmail($email))
 134:     {
 135:         addAlert("danger", lang("ACCOUNT_INVALID_EMAIL"));
 136:         $error_count++;
 137:     }
 138:     if(minMaxRange(1,150,$title)) {
 139:         addAlert("danger", lang("ACCOUNT_TITLE_CHAR_LIMIT",array(1,150)));
 140:         $error_count++;
 141:     }
 142:     if(minMaxRange(8,50,$password) && minMaxRange(8,50,$passwordc))
 143:     {
 144:         addAlert("danger", lang("ACCOUNT_PASS_CHAR_LIMIT",array(8,50)));
 145:         $error_count++;
 146:     }
 147:     else if($password != $passwordc)
 148:     {
 149:         addAlert("danger", lang("ACCOUNT_PASS_MISMATCH"));
 150:         $error_count++;
 151:     }
 152: 
 153:     if(usernameExists($user_name)) {
 154:         addAlert("danger", lang("ACCOUNT_USERNAME_IN_USE",array($user_name)));
 155:         $error_count++;
 156:     }
 157:     if(displayNameExists($display_name)) {
 158:         addAlert("danger", lang("ACCOUNT_DISPLAYNAME_IN_USE",array($display_name)));
 159:         $error_count++;
 160:     }
 161:     if(emailExists($email)) {
 162:         addAlert("danger", lang("ACCOUNT_EMAIL_IN_USE",array($email)));
 163:         $error_count++;
 164:     }
 165: 
 166:     //Construct a secure hash for the plain text password
 167:     $password_hash = passwordHashUF($password);
 168:     if ($password_hash === null){
 169:         addAlert("danger", lang("PASSWORD_HASH_FAILED"));
 170:         $error_count++;        
 171:     }
 172:     
 173:     // Exit on any invalid parameters
 174:     if($error_count != 0)
 175:         return false;
 176:     
 177: 
 178:     //Construct a unique activation token (even if activation is not required)
 179:     $activation_token = generateActivationToken();
 180:     $active = 1;
 181: 
 182:     //Do we need to require that the user activate their account first?
 183:     if($require_activation) {
 184: 
 185:         //User must activate their account first
 186:         $active = 0;
 187: 
 188:         $mailSender = new userCakeMail();
 189: 
 190:         //Build the activation message
 191:         $activation_message = lang("ACCOUNT_ACTIVATION_MESSAGE",array(SITE_ROOT . "api/", $activation_token));
 192: 
 193:         //Define more if you want to build larger structures
 194:         $hooks = array(
 195:             "searchStrs" => array("#ACTIVATION-MESSAGE","#ACTIVATION-KEY","#USERNAME#"),
 196:             "subjectStrs" => array($activation_message,$activation_token,$display_name)
 197:         );
 198: 
 199:         /* Build the template - Optional, you can just use the sendMail function
 200:         Instead to pass a message. */
 201:         // If there is a mail failure, fatal error
 202:         if(!$mailSender->newTemplateMsg("new-registration.txt",$hooks)) {
 203:             addAlert("danger", lang("MAIL_ERROR"));
 204:             return false;
 205:         } else {
 206:             //Send the mail. Specify users email here and subject.
 207:             //SendMail can have a third paremeter for message if you do not wish to build a template.
 208:             if(!$mailSender->sendMail($email, "Please activate your account")) {
 209:                 addAlert("danger", lang("MAIL_ERROR"));
 210:                 return false;
 211:             }
 212:         }
 213:     }
 214: 
 215:     // Insert the user into the database and return the new user's id
 216:     return addUser($user_name, $display_name, $title, $password_hash, $email, $active, $activation_token);
 217: }
