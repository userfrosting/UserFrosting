160     /**
161      * Generate the full list of paths for a particular group of includes, including common includes.
162      *
163      * This function merges any group-specific includes with the common includes.
164      * It also prepends the root URL for the type of resource (JS or CSS) to those includes.
165      * @param string $root the root URL for this type of resource (usually the JS or CSS URL).
166      * @param array $raw the raw array containing all includes, from which to extract the appropriate includes.
167      * @param string $group_name the name of the page group to generate the list of includes for.
168      * @param string $minfile the name of the minfile to create, with path relative to the root folder for this type of asset.
169      * @param bool $minify specify whether or not to minify and combine the includes.  If set to `true`, the returned array will have the minified combined asset file instead of the unminified files.
170      * @param bool $include_externals specify whether or not to include non-local assets as well (e.g., CDN)
171      * @return array an array containing the full paths of the files to be included for the specified asset type.
172      */
173     private function mergeIncludes($root, $raw, $group_name, $minfile, $minify = false, $include_externals = true){
174         // Combine the common and group-specific includes    
175         if (isset($raw["common"]))
176             $includes = $raw["common"];
177         else
178             $includes = [];
179         if ($group_name != "common" && isset($raw[$group_name]))
180             $includes = array_merge($includes, $raw[$group_name]);
181  
182         // For minified, replace with minified file but we still need to include any external includes
183         if ($minify){
184             $includes_parsed = [$root . "/" . $minfile];
185             // Include external files
186             if ($include_externals) {
187                 foreach ($includes as $path){
188                     if (strpos($path, 'http') === 0)
189                         $includes_parsed[] = $path;
190                 }
191             }
192         } else {        
193             $includes_parsed = [];
194             foreach ($includes as $path){
195                 if (strpos($path, 'http') === 0) {
196                     if ($include_externals)
197                         $includes_parsed[] = $path;
198                 } else
199                     $includes_parsed[] = $root . "/" . $path;
200             }
201         }
202         return $includes_parsed;
203     }

205     /**
206      * Builds the minified CSS and JS files for the site.
207      *
208      * This function uses the Yahoo User Interface (YUI) interface (http://yui.github.io/yuicompressor/) to compress and minify the files as specified in this PageSchema object.  The user account under which the web server runs must have write access to the `min` subdirectories.
209      * @param bool $debug If set to true, write information about the minification process to the error log.
210      * @return void
211      */
212     public function build($debug = false){
213         // Determine path to YUI jar file
214         $yui = __DIR__ . "/yuicompressor-2.4.8.jar";      
215         
216         /*
217          * In XAMPP and Mac OSX, by default Apache is run under the user 'daemon'.  To grant proper write permissions, run the following shell command:
218          * `sudo chown -R daemon:daemon <write_path>`
219          * This will grant ownership of the path to the web server.
220          */
221         $web_user = trim(`whoami`);
222          
223         if ($debug) {
224             error_log("Running as user: $web_user");
225             error_log("YUI compressor is in $yui");
226         }
227         
228         // For each manifest group, build the corresponding minified, bundled (concatenated) JS and CSS files
229         
230         // Build CSS
231         foreach ($this->_css_includes as $group_name => $group){
232             if ($debug) {
233                 error_log("Building CSS for page group '$group_name'...");
234             }
235             $this->buildGroup($yui, $this->_css_path, $this->_css_includes, $group_name, "$group_name.min.css", $debug);
236         }
237         
238         // Build head JS
239         foreach ($this->_js_includes_top as $group_name => $group){
240             if ($debug) {
241                 error_log("Building top JS for page group '$group_name'...");
242             }
243             $this->buildGroup($yui, $this->_js_path, $this->_js_includes_top, $group_name, "$group_name-top.min.js", $debug);
244         }
245 
246         // Build footer JS
247         foreach ($this->_js_includes_bottom as $group_name => $group){
248             if ($debug) {
249                 error_log("Building bottom JS for page group '$group_name'...");
250             }
251             $this->buildGroup($yui, $this->_js_path, $this->_js_includes_bottom, $group_name, "$group_name-bottom.min.js", $debug);
252         }
253     }

255     /**
256      * Build the minified file for a specified page group and asset type.
257      *
258      * Uses the YUI compressor (http://yui.github.io/yuicompressor/) to minify JS and CSS asset files, writing to a specified file.
259      * @param string $yui the path to the YUI compressor JAR.
260      * @param string $path_root the root file path for this type of resource (usually the JS or CSS path).
261      * @param array $raw the raw array containing all includes, from which to extract the appropriate includes.
262      * @param string $group_name the name of the page group to generate the minified file for.
263      * @param string $minfile the name of the minfile to create, with path relative to the root folder for this type of asset.
264      * @param bool $debug if set to true, write information about the minification process to the error log.
265      * @return void
266      */  
267     private function buildGroup($yui, $path_root, $includes, $group_name, $minfile, $debug = false){
268         $paths = $this->mergeIncludes($path_root, $includes, $group_name, "", false, false);
269         
270         // Test permissions on writing to min file:
271         $output_dir = $path_root . "/min/";
272         $output_file = $output_dir . $minfile;
273         $this->writeMinifiedFile($output_dir, $output_file, []);
274 
275         // Each file will be minified, and the result appended to the output array
276         $group_arr = [];
277         foreach($paths as $path){
278             if ($debug) {
279                 error_log("----Minifying file '$path'");
280             }
281             
282             // This will let us run the compressor and detect any errors
283             $descriptorspec = array(
284                 0 => array("pipe", "r"),  // stdin
285                 1 => array("pipe", "w"),  // stdout
286                 2 => array("pipe", "w"),  // stderr
287             );
288             
289             $process = proc_open("export DYLD_LIBRARY_PATH=''; java -jar $yui $path", $descriptorspec, $pipes);
290             
291             $stdout = stream_get_contents($pipes[1]);
292             fclose($pipes[1]);
293             
294             $stderr = stream_get_contents($pipes[2]);
295             fclose($pipes[2]);
296             
297             // Throw exception if there was an error
298             if ($stderr){
299                 throw new \Exception("Unable to minify '$path'.  YUI error message:\n" . $stderr);
300             }
301             
302             // If successful, append this minified content to the group file
303             $group_arr[] = $stdout;
304         }
305         
306         // Write minified CSS to the output file
307         if ($debug) {
308             error_log("--Attempting to write to file '$output_file'...");
309         }
310         return $this->writeMinifiedFile($output_dir, $output_file, $group_arr);  
311     }

313     /**
314      * Write minified content to a specified file.
315      *
316      * @param string $output_dir the directory that will contain the minified file.
317      * @param string $output_file the full path and filename of the minified file to write.
318      * @param array $content an array of strings that makes up the content to be written.
319      * @throws Exception Insufficient permissions to write to the specified directory.
320      * @return void
321      */
322     private function writeMinifiedFile($output_dir, $output_file, $content){
323         try {
324             file_put_contents($output_file, implode("\n", $content));
325             return true;
326         } catch (\Exception $e){
327             $web_user = trim(`whoami`);
328             throw new \Exception("Access denied to write to '$output_file'.  Be sure that the directory exists, and that the user '$web_user' has permission to write to this directory.  To grant write access in OSX/Linux/etc, try running the chown command: 'sudo chown -R $web_user:$web_user $output_dir'.");
329         }
330     }
